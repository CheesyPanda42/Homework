5 /* opaque pointer since definition of Node is hidden from the client we need to
6  * provide some way to use Node struct, and here it is -- typedef. At this
7  * point Node is NOT KNOWN to the compiler (which makes Node an INCOMPLETE
8  * TYPE, but declaring a pointer to an incomplete type is allowed by C (and
9  * C++).
10  *
11  * see http://en.wikipedia.org/wiki/Opaque_pointer
12  *
13  * we also need a special type ConstElementNode_handle:
14  * notice that:
15  * even though "ElementNode*" is same as "ElementNode_handle"
16  * "const ElementNode*" and "const ElementNode_handle"
17  * are DIFFERENT:
18  * const ElementNode* p; is modifiable pointer to constant data
19  * while
20  * const ElementNode_handle p; is constant pointer to modifiable data
21  */
22 typedef struct ElementNode * ElementNode_handle;
23 typedef struct ElementNode const * ConstElementNode_handle;
